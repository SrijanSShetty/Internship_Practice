To compute gross salary 
Step 1:start
Step 2:declare b_salary,g_salary;
Step 3:read b_salary;
Step 4:if b_salary <= 10000
        g_salary=b_salary + (b_salary*(20/100))+(b_salary*(80/100))
        display g_salary
Step 5:if b_salary <= 20000
        g_salary=b_salary + (b_salary*(25/100)) + (b_salary*(90/100))
        display g_salary
Step 6:else
      g_salary=b_salary + (b_salary*(30/100)) + (b_salary*(95/100))
      display g_salary
Step 7:stop
